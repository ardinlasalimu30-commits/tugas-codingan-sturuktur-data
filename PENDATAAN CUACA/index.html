01
Gambaran Integrasi Dua Sistem Queue

Prinsip FIFO (First In, First Out): data yang masuk pertama ke antrian, diproses pertama. Seperti antrian dermaga — kapal yang datang lebih awal, berangkat lebih awal. Dua Queue berjalan bersama dan saling mengirim informasi keputusan.

Skor Risiko Status Cuaca Keputusan Melaut Faktor Pemicu
0.00 – 0.29 AMAN ✅ BERANGKAT Angin <15km/h, Gelombang <0.8m, Hujan ringan
0.30 – 0.59 WASPADA ⚠️ TUNDA Angin 15–40km/h, Gelombang 0.8–2.5m, Hujan sedang

// ── Status keputusan melaut (enum) ──────────────────
enum StatusMelaut { BERANGKAT, TUNDA, BATAL };

// ── Satu data pembacaan sensor cuaca ────────────────
struct DataCuaca {
    string timestamp;
    float  suhu;          // °C
    float  kelembaban;    // %
    float  angin_kmh;     // km/jam
    float  gelombang_m;   // meter
    float  hujan_mmh;     // mm/jam
    string lokasi;
    float  skor_risiko;   // 0.0 = aman, 1.0 = bahaya
    string status_risiko; // "AMAN" | "WASPADA" | "BAHAYA"
};

// ── Satu jadwal nelayan ──────────────────────────────
struct JadwalMelaut {
    string       id_jadwal;
    string       nama_nelayan;
    string       nama_kapal;
    int          jumlah_abk;
    string       jam_daftar;
    string       jam_rencana;
    string       tujuan;
    StatusMelaut status;      // default: TUNDA
};
QueueCuaca — implementasi manual dengan linked list
// ── Node linked list untuk Queue Cuaca ──────────────
struct NodeCuaca {
    DataCuaca  data;
    NodeCuaca* next;
    NodeCuaca(DataCuaca d) : data(d), next(nullptr) {}
};

class QueueCuaca {
private:
    NodeCuaca* head;   // depan — DEQUEUE dari sini
    NodeCuaca* tail;   // belakang — ENQUEUE ke sini
    int        ukuran;
    int        kapasitas;

public:
    QueueCuaca(int kap = 50)
        : head(nullptr), tail(nullptr),
          ukuran(0), kapasitas(kap) {}

    // ENQUEUE: tambah node baru ke TAIL
    bool enqueue(DataCuaca data) {
        if (ukuran >= kapasitas) return false;
        NodeCuaca* baru = new NodeCuaca(data);
        if (tail == nullptr)
            head = tail = baru;          // queue kosong
        else {
            tail->next = baru;           // sambung ke tail lama
            tail = baru;                 // geser tail ke node baru
        }
        ukuran++;
        return true;
    }

    // DEQUEUE: hapus dan kembalikan node dari HEAD (FIFO)
    DataCuaca dequeue() {
        if (kosong()) throw runtime_error("Queue kosong!");
        NodeCuaca* temp = head;
        DataCuaca  data = temp->data;
        head = head->next;               // geser head ke node berikut
        if (head == nullptr) tail = nullptr;
        delete temp;                     // bebaskan memori
        ukuran--;
        return data;
    }

    // PEEK: lihat HEAD tanpa menghapus
    DataCuaca peek() const {
        if (kosong()) throw runtime_error("Queue kosong!");
        return head->data;
    }

    bool kosong()    const { return ukuran == 0; }
    int  getUkuran() const { return ukuran; }
};
cuaca.cpp — bacaSensor, validasi, hitungRisiko
// Simulasi baca sensor (nyata: dari serial port / API BMKG)
DataCuaca bacaSensor(const string& lokasi) {
    DataCuaca d;
    d.timestamp   = waktuSekarang();
    d.lokasi      = lokasi;
    d.suhu        = 26.0f + (rand() % 120) / 10.0f;
    d.kelembaban  = 60.0f + (rand() % 380) / 10.0f;
    d.angin_kmh   = 5.0f  + (rand() % 560) / 10.0f;
    d.gelombang_m = 0.2f  + (rand() % 43)  / 10.0f;
    d.hujan_mmh   = 0.0f  + (rand() % 300) / 10.0f;
    return d;
}

// Validasi: tolak nilai sensor yang tidak wajar
bool validasiCuaca(const DataCuaca& d) {
    if (d.suhu < -10 || d.suhu > 60)         return false;
    if (d.kelembaban < 0 || d.kelembaban > 100) return false;
    if (d.angin_kmh < 0 || d.angin_kmh > 200)   return false;
    if (d.gelombang_m < 0 || d.gelombang_m > 20) return false;
    return true;
}

// Hitung skor risiko 0.0 – 1.0
float hitungRisiko(DataCuaca& d) {
    float skor = 0.0f;

    // Kontribusi angin (maks 0.35)
    if      (d.angin_kmh > 40) skor += 0.35f;
    else if (d.angin_kmh > 25) skor += 0.18f;
    else if (d.angin_kmh > 15) skor += 0.08f;

    // Kontribusi gelombang (maks 0.40)
    if      (d.gelombang_m > 2.5f) skor += 0.40f;
    else if (d.gelombang_m > 1.5f) skor += 0.22f;
    else if (d.gelombang_m > 0.8f) skor += 0.10f;

    // Kontribusi hujan (maks 0.25)
    if      (d.hujan_mmh > 20) skor += 0.25f;
    else if (d.hujan_mmh > 10) skor += 0.12f;

    d.skor_risiko  = min(skor, 1.0f);
    d.status_risiko = (skor < 0.3f) ? "AMAN"
                    : (skor < 0.6f) ? "WASPADA"
                    :                   "BAHAYA";
    return d.skor_risiko;
}
jadwal.cpp — buatJadwal, nentukanStatus, kirimNotifikasi
int idCounter = 1;

// Buat objek jadwal nelayan baru
JadwalMelaut buatJadwal(
    const string& nama, const string& kapal,
    int abk, const string& tujuan,
    const string& jam_rencana
) {
    JadwalMelaut j;
    j.id_jadwal    = "JDL-" + to_string(idCounter++);
    j.nama_nelayan = nama;
    j.nama_kapal   = kapal;
    j.jumlah_abk   = abk;
    j.jam_daftar   = waktuSekarang();
    j.jam_rencana  = jam_rencana;
    j.tujuan       = tujuan;
    j.status       = TUNDA;   // default sebelum dicek cuaca
    return j;
}

// Konversi skor risiko → keputusan melaut
StatusMelaut nentukanStatus(float skor) {
    if      (skor < 0.30f) return BERANGKAT;
    else if (skor < 0.60f) return TUNDA;
    else                    return BATAL;
}

// Cetak notifikasi ke layar / perangkat output
void kirimNotifikasi(const JadwalMelaut& j) {
    switch (j.status) {
        case BERANGKAT:
            cout << "[OK] Aman berangkat ke " << j.tujuan << "\n";
            break;
        case TUNDA:
            cout << "[!!] Waspada. Tunda keberangkatan.\n";
            break;
        case BATAL:
            cout << "[XX] BERBAHAYA! Dilarang melaut!\n";
            break;
    }
}
main.cpp — integrasi dua queue
int main() {
    srand(time(nullptr));

    // Inisialisasi dua queue
    QueueCuaca  qCuaca(100);
    QueueJadwal qJadwal(30);

    // ─ 1. Daftarkan nelayan ke queue jadwal ─
    qJadwal.enqueue(buatJadwal(
        "Pak Umar", "KM Harapan", 4,
        "Perairan Buton", "05:00"
    ));
    qJadwal.enqueue(buatJadwal(
        "Pak Saleh", "KM Barokah", 6,
        "Laut Banda", "05:30"
    ));

    // ─ 2. Baca sensor & masukkan ke queue cuaca ─
    DataCuaca d = bacaSensor("Teluk Kendari");
    if (validasiCuaca(d)) {
        hitungRisiko(d);
        qCuaca.enqueue(d);
    }

    // ─ 3. Proses cuaca dari HEAD ─
    DataCuaca cuaca = qCuaca.dequeue();

    // ─ 4. Proses semua jadwal berdasarkan cuaca terkini ─
    while (!qJadwal.kosong()) {
        JadwalMelaut j = qJadwal.dequeue();
        j.status = nentukanStatus(cuaca.skor_risiko);
        cetakJadwal(j);
        kirimNotifikasi(j);
    }

    return 0;
}
